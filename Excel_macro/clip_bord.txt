Sub sample()
    Dim buf As String
    Dim j As Integer
    Dim CB As New DataObject
    j = 1
    CB.GetFromClipboard
    buf2 = CB.GetText
    
    Do While True
      buf = CB.GetText
      Sleep 100
      If Not buf = buf2 Then
        Cells(j, 1).Value = buf
        j = j + 1
        buf2 = buf
      End If
      DoEvents
    Loop
End Sub
